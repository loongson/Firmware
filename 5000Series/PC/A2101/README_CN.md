	Serial Number: MX20BA001J
	Asset Tag: Type2 - Board Asset Tag

		Board is a hosting board
		Board is replaceable
	Location In Chassis: Type2 - Board Chassis Location
	Chassis Handle: 0xFFFF
	Type: Motherboard
	Contained Object Handles: 0
